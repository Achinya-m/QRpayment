SELECT
    DATE_FORMAT(o.created_at, '%d/%m %H:00') AS timeLabel,

    CASE
        WHEN '{{ SelectShop.selectedOptionValue || "" }}' = ''
        THEN s.shop_name
        ELSE d.device_id
    END AS seriesName,

    SUM(o.amount) AS totalAmount

FROM mpete_vending.orders o

LEFT JOIN mpete_vending.shops s
    ON o.shop_key = s.shop_key

LEFT JOIN mpete_vending.devices d
    ON o.device_id = d.device_id

WHERE
    o.merchant_key = '{{ keyMerchant.data?.[0]?.merchant_key || "" }}'
    AND o.status IN ('paid', 'late_paid')

    AND (
        '{{ SelectShop.selectedOptionValue || "" }}' = ''
        OR o.shop_key = '{{ SelectShop.selectedOptionValue || "" }}'
    )

GROUP BY
    timeLabel,
    seriesName

ORDER BY
    STR_TO_DATE(timeLabel, '%d/%m %H:00') ASC;
