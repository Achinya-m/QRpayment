SELECT 
    CONCAT('฿ ', FORMAT(SUM(o.amount), 0)) AS totalRevenueDay,
    FORMAT(COUNT(o.order_id), 0) AS totalTransactionsDay
FROM 
    mpete_vending.orders o
LEFT JOIN 
    mpete_vending.shops s ON o.shop_key = s.shop_key
WHERE 
		o.status IN ('paid', 'late_paid')
		AND DATE(o.created_at) = CURDATE()
		AND (
        {{ !SelectShop.selectedOptionValue }} 
        OR o.shop_key = '{{ SelectShop.selectedOptionValue }}'
    );
