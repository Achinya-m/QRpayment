SELECT
    FORMAT(COALESCE(SUM(o.amount),0), 0) AS totalRevenueDay,
    FORMAT(COALESCE(COUNT(o.order_id),0), 0) AS totalTransactionsDay

FROM mpete_vending.orders o
JOIN mpete_vending.merchants m 
    ON o.merchant_key = m.merchant_key

WHERE o.merchant_key = '{{ keyMerchant.data?.[0]?.merchant_key || "" }}'
AND o.status IN ('paid', 'late_paid')
AND DATE(o.created_at) = CURDATE()

AND (
    '{{ SelectShop.selectedOptionValue || "" }}' = ''
    OR o.shop_key = '{{ SelectShop.selectedOptionValue || "" }}'
);
